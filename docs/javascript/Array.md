``` js
// 创建数组
new Array(3) // 3表示数组长度
var arr = new Array(3, '4', true) // 3, '4', true 表示数组内容
var arr = [3, '4', true] // 3, '4', true 表示数组内容
console.log(arr[0]) // 读取数组下标对应的值
console.log(arr.length) // 返回数组的长度，也等于最后一个下标 + 1
----------------------------------------------------------------------------------------------
// 遍历数组
for(var i = 0; i < arr.length; i++) { arr[i] }
----------------------------------------------------------------------------------------------
// 栈方法
arr.push(1) // 把1添加到arr的尾部，返回新数组的长度Number
arr.unshift(1) // 把1添加到arr的头部，返回新数组的长度Number
arr.pop() // 删除数组最后一个元素，返回被删除的元素
arr.shift() // 删除数组第一个元素，返回被删除的元素
----------------------------------------------------------------------------------------------
// 操作数组
arr.sort(function(a, b) { return b - a }) // 降序
arr.sort(function(a, b) { return a - b }) // 升序
arr3 = arr1.concat(arr2) //用于连接两个或多个数组
arr.join(',') // 将数组转换成字符串，返回转换后的字符串，默认用逗号分隔，通过参数改变分隔符
arr.reverse() // 数组反序，返回转换后的数组
arr.slice(0, 3) // 数组截取，返回截取的数组，从下标0开始截取到3结束，如没传3则表示从0截取到最后，参数如有负数则以arr.length + 负数再计算
arr.splice(2, 2) // 数组删除，返回被删除的元素所组成的数组，删除arr下标为2的后2个元素
arr.splice(2, 0, 3) // 数组插入，返回一个空数组（没有被删除的元素），在下标为2的后面插入3
arr.splice(2, 2, 3) // 数组替换，返回被删除的元素所组成的数组，在下标为2的后面删除2个元素，再插入3
arr.indexOf(7, 3) // 数组查找，返回下标，查找7在数组中从下标3开始找位置，如果找不到返回 -1
arr.lastIndexOf(7, 3) // 数组查找，返回下标，从后往前找，查找7在数组中从下标3开始找位置，如果找不到返回 -1
```
